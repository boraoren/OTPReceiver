# OTPReceiver
OTP Receiver Without SMS Permission in Android Application

#Add it in your root build.gradle at the end of repositories:

	allprojects {
		repositories {
			...
			maven { url 'https://jitpack.io' }
		}
	}
  
 #Add the dependency
 
 	dependencies {
	        implementation 'com.github.mobileappsnk:OTPReceiver:v1.1'
	}
